<?php get_header(); ?>

<div class="courses-grid">
    <?php if ( have_posts() ) : while ( have_posts() ) : the_post(); ?>
        <div class="course-box">
            <h2 class="course-title"><a href="<?php the_permalink(); ?>"><?php the_title(); ?></a></h2>
            
            <?php
            $instructor = get_post_meta(get_the_ID(), 'instructor', true);
            $duration   = get_post_meta(get_the_ID(), 'duration', true);
            $level      = get_post_meta(get_the_ID(), 'level', true);
            ?>
            
            <?php if ($instructor) : ?><p><strong>Instructor:</strong> <?php echo esc_html($instructor); ?></p><?php endif; ?>
            <?php if ($duration) : ?><p><strong>Duration:</strong> <?php echo esc_html($duration); ?></p><?php endif; ?>
            <?php if ($level) : ?><p><strong>Level:</strong> <?php echo esc_html($level); ?></p><?php endif; ?>
            
            <div class="course-actions">
                <a href="<?php the_permalink(); ?>" class="access-course-btn">Access Course</a>
            </div>
        </div>
    <?php endwhile; else: ?>
        <p>No courses found.</p>
    <?php endif; ?>
</div>

<?php get_footer(); ?>
